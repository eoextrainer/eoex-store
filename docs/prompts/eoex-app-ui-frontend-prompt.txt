// Email validation
const form = document.querySelector('.signup');
const emailInput = document.querySelector('#email');
const errorEl = document.querySelector('.signup .error');

form.addEventListener('submit', (e) => {
  e.preventDefault();
  const email = emailInput.value.trim();
  const isValid = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
  if (!isValid) {
    errorEl.textContent = 'Please enter a valid email address.';
    emailInput.setAttribute('aria-invalid', 'true');
    emailInput.focus();
  } else {
    errorEl.textContent = '';
    emailInput.removeAttribute('aria-invalid');
    alert('Thank you!'); // replace with real handling
  }
});

// Carousel
const track = document.querySelector('.track');
const prevBtn = document.querySelector('.prev');
const nextBtn = document.querySelector('.next');

let cardWidth = 0;
function updateCardWidth() {
  const firstCard = track.querySelector('.card');
  if (firstCard) {
    cardWidth = firstCard.getBoundingClientRect().width + parseFloat(getComputedStyle(track).gap);
  }
}
function scrollByCards(dir = 1) {
  track.parentElement.scrollBy({ left: dir * cardWidth * 3, behavior: 'smooth' });
}
window.addEventListener('resize', () => { updateCardWidth(); });
updateCardWidth();

prevBtn.addEventListener('click', () => scrollByCards(-1));
nextBtn.addEventListener('click', () => scrollByCards(1));

// Accordion
document.querySelectorAll('.accordion .item').forEach((item) => {
  const btn = item.querySelector('.trigger');
  const panel = item.querySelector('.panel');

  btn.addEventListener('click', () => {
    const open = item.classList.toggle('is-open');
    btn.setAttribute('aria-expanded', open ? 'true' : 'false');

    // Close others
    document.querySelectorAll('.accordion .item').forEach((other) => {
      if (other !== item) {
        other.classList.remove('is-open');
        other.querySelector('.trigger').setAttribute('aria-expanded', 'false');
      }
    });
  });
});

jsx
import React, { useState } from 'react';
import { View, Text, TouchableOpacity, LayoutAnimation, StyleSheet } from 'react-native';

const faqs = [
  { q: 'What is StreamNow?', a: 'Stream a wide variety of shows and movies on your devices.' },
  { q: 'How much does it cost?', a: 'Plans start at ‚Ç¨7.99 per month.' },
];

export default function FAQ() {
  const [openIndex, setOpenIndex] = useState(null);

  const toggle = (i) => {
    LayoutAnimation.configureNext(LayoutAnimation.Presets.easeInEaseOut);
    setOpenIndex(openIndex === i ? null : i);
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Frequently Asked Questions</Text>
      {faqs.map((item, i) => (
        <View key={i} style={styles.item}>
          <TouchableOpacity onPress={() => toggle(i)}>
            <Text style={styles.question}>{item.q}</Text>
          </TouchableOpacity>
          {openIndex === i && <Text style={styles.answer}>{item.a}</Text>}
        </View>
      ))}
    </View>
  );
}

const styles = StyleSheet.create({
  container: { padding: 16 },
  title: { color: '#fff', fontSize: 20, fontWeight: '700', marginBottom: 12 },
  item: { marginBottom: 12 },
  question: { color: '#fff', fontSize: 16, fontWeight: '600' },
  answer: { color: '#B3B3B3', marginTop: 6 },

WebBrowser.maybeCompleteAuthSession();

const googleConfig = {
  clientId: '<YOUR_GOOGLE_OAUTH_CLIENT_ID>',
  responseType: 'token',
  scopes: ['profile', 'email'],
  redirectUri: AuthSession.makeRedirectUri({ useProxy: true }),
};

export default function SignIn({ auth }) {
  const [email, setEmail] = React.useState('');
  const [password, setPassword] = React.useState('');

  const onLocal = async () => {
    try {
      await auth.signInLocal(email, password);
    } catch (e) {
      Alert.alert('Sign-in failed', e.message);
    }
  };

  const onGoogle = async () => {
    const authUrl = `https://accounts.google.com/o/oauth2/v2/auth?client_id=${googleConfig.clientId}&redirect_uri=${encodeURIComponent(googleConfig.redirectUri)}&response_type=${googleConfig.responseType}&scope=${encodeURIComponent(googleConfig.scopes.join(' '))}`;
    const result = await AuthSession.startAsync({ authUrl });
    if (result.type === 'success' && result.params?.access_token) {
      // For demo: you would fetch profile from Google APIs using the token
      auth.signInGoogle({ email: 'google_user@example.com' });
    } else {
      Alert.alert('Google sign-in cancelled');
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Welcome back</Text>
      <TextInput
        style={styles.input}
        placeholder="Email"
        placeholderTextColor="#B3B3B3"
        autoCapitalize="none"
        keyboardType="email-address"
        value={email}
        onChangeText={setEmail}

export async function openYouTube(videoIdOrUrl) {
  const id = videoIdOrUrl.includes('youtube.com')
    ? new URL(videoIdOrUrl).searchParams.get('v')
    : videoIdOrUrl;

  }
  return Linking.openURL(webUrl);
}

Session manager
jsx
// src/services/session.js
import AsyncStorage from '@react-native-async-storage/async-storage';
import * as SecureStore from 'expo-secure-store'; // Expo; or use react-native-keychain in bare RN

const SESSION_KEY = 'streamnow.session';
const TOKEN_KEY = 'streamnow.token';

export async function saveSession(profile) {
  await AsyncStorage.setItem(SESSION_KEY, JSON.stringify(profile));
}
export async function loadSession() {
  const raw = await AsyncStorage.getItem(SESSION_KEY);
  return raw ? JSON.parse(raw) : null;
}
export async function clearSession() {
  await AsyncStorage.removeItem(SESSION_KEY);
  await SecureStore.deleteItemAsync(TOKEN_KEY);
}

export async function saveToken(token) {
  await SecureStore.setItemAsync(TOKEN_KEY, token);
}
export async function getToken() {
  return SecureStore.getItemAsync(TOKEN_KEY);
}

      const existing = await loadSession();
      if (existing) setUser(existing);

      await clearSession();
      setUser(null);

    user,
  }), [user]);

    return (
      <View style={{ flex: 1, backgroundColor: '#141414', justifyContent: 'center', alignItems: 'center' }}>
        <ActivityIndicator color="#E50914" />
      </View>

      {/* stack with SignIn/Home/Player */}
    </NavigationContainer>
  );

const palettes = {

export const ThemeContext = createContext({ theme: palettes.dark });

export function ThemeProvider({ children, mode }) {
  const scheme = useColorScheme(); // 'light' | 'dark' | null
  const resolved = mode || scheme || 'dark';
  const value = useMemo(() => ({ theme: palettes[resolved], mode: resolved }), [resolved]);
  return <ThemeContext.Provider value={value}>{children}</ThemeContext.Provider>;
}

Here‚Äôs how you can persist the theme choice so that when a user toggles between light and dark, their preference is remembered across app restarts. We‚Äôll use SecureStore for sensitive values if you want extra protection, but since theme mode isn‚Äôt sensitive, AsyncStorage is usually sufficient. I‚Äôll show you both so you can decide.

üîß Setup
Install dependencies:

bash
npm install @react-native-async-storage/async-storage
expo install expo-secure-store   # if using Expo
üé® ThemeContext with persistence
jsx
// src/theme/ThemeContext.js
import React, { createContext, useState, useEffect, useMemo } from 'react';
import { useColorScheme } from 'react-native';
import AsyncStorage from '@react-native-async-storage/async-storage';
import * as SecureStore from 'expo-secure-store';

const palettes = {
  dark: {
    bg: '#141414',
    panel: '#221F1F',
    text: '#FFFFFF',
    subtext: '#B3B3B3',
    red: '#E50914',
  },
  light: {
    bg: '#FFFFFF',
    panel: '#F3F3F3',
    text: '#000000',
    subtext: '#4A4A4A',
    red: '#E50914',
  },
};

export const ThemeContext = createContext();

const STORAGE_KEY = 'streamnow.theme';

export function ThemeProvider({ children }) {
  const systemScheme = useColorScheme(); // 'light' | 'dark'
  const [mode, setMode] = useState(systemScheme || 'dark');
  const [booted, setBooted] = useState(false);

  // Load persisted theme
  useEffect(() => {
    (async () => {
      try {
        const stored = await AsyncStorage.getItem(STORAGE_KEY);
        if (stored) {
          setMode(stored);
        } else {
          // fallback to system scheme
          setMode(systemScheme || 'dark');
        }
      } catch (e) {
        console.warn('Failed to load theme', e);
      }
      setBooted(true);
    })();
  }, [systemScheme]);

  // Save theme choice
  const persistMode = async (newMode) => {
    setMode(newMode);
    try {
      await AsyncStorage.setItem(STORAGE_KEY, newMode);
      // Optionally also store in SecureStore
      await SecureStore.setItemAsync(STORAGE_KEY, newMode);
    } catch (e) {
      console.warn('Failed to save theme', e);
    }
  };

  const value = useMemo(() => ({
    theme: palettes[mode],
    mode,
    setMode: persistMode,
  }), [mode]);

  if (!booted) return null; // or splash screen

  return (
    <ThemeContext.Provider value={value}>
      {children}
    </ThemeContext.Provider>
  );
}

To make the cascade feel more fluid, you can combine stagger with parallel animations so that sections overlap slightly instead of waiting for one to finish before the next begins. The trick is to use Animated.stagger with a shorter delay than the animation duration, and group each section‚Äôs fade/slide/scale in a parallel.

üîß Example: Hero + Carousel overlap

Here‚Äôs how you can stagger with different delays per section so the rhythm feels more organic. Instead of a uniform stagger, each section gets its own delay before starting its parallel animation group.

üîß Implementation

// src/theme/ThemeContext.js
import React, { createContext, useState, useEffect, useMemo } from 'react';
import { useColorScheme } from 'react-native';
import AsyncStorage from '@react-native-async-storage/async-storage';
import * as SecureStore from 'expo-secure-store';

const palettes = {
  dark: { bg: '#141414', panel: '#221F1F', text: '#FFFFFF', subtext: '#B3B3B3', red: '#E50914' },
  light: { bg: '#FFFFFF', panel: '#F3F3F3', text: '#000000', subtext: '#4A4A4A', red: '#E50914' },
};

export const ThemeContext = createContext();
const THEME_KEY = 'streamnow.theme';

export function ThemeProvider({ children }) {
  const system = useColorScheme();
  const [mode, setMode] = useState(system || 'dark');
  const [booted, setBooted] = useState(false);

  useEffect(() => {
    (async () => {
      const stored = await AsyncStorage.getItem(THEME_KEY);
      setMode(stored || system || 'dark');
      setBooted(true);
    })();
  }, [system]);

  const persistMode = async (next) => {
    setMode(next);
    await AsyncStorage.setItem(THEME_KEY, next);
    await SecureStore.setItemAsync(THEME_KEY, next); // optional
  };

  const value = useMemo(() => ({ theme: palettes[mode], mode, setMode: persistMode }), [mode]);

  if (!booted) return null;
  return <ThemeContext.Provider value={value}>{children}</ThemeContext.Provider>;
}

import React from 'react';
import { View, Text, TextInput, TouchableOpacity, StyleSheet, Alert } from 'react-native';
import * as AuthSession from 'expo-auth-session';

export default function SignIn({ auth }) {
  const [email, setEmail] = React.useState('');
  const [password, setPassword] = React.useState('');

  const onLocal = async () => {
    try { await auth.signInLocal(email, password); } catch (e) { Alert.alert('Sign-in failed', e.message); }
  };

  const googleClientId = '<YOUR_GOOGLE_OAUTH_CLIENT_ID>';
  const redirectUri = AuthSession.makeRedirectUri({ useProxy: true });
  const authUrl = `https://accounts.google.com/o/oauth2/v2/auth?client_id=${googleClientId}&redirect_uri=${encodeURIComponent(redirectUri)}&response_type=token&scope=${encodeURIComponent('profile email')}`;

  const onGoogle = async () => {
    const res = await AuthSession.startAsync({ authUrl });
    if (res.type === 'success' && res.params?.access_token) {
      const info = await fetch('https://www.googleapis.com/oauth2/v2/userinfo', {
        headers: { Authorization: `Bearer ${res.params.access_token}` },
      }).then(r => r.json());
      await auth.signInGoogle({ email: info.email, name: info.name, provider: 'google' }, res.params.access_token);
    } else {
      Alert.alert('Google sign-in cancelled');
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Welcome back</Text>
      <TextInput style={styles.input} placeholder="Email" placeholderTextColor="#B3B3B3" value={email} onChangeText={setEmail} />
      <TextInput style={styles.input} placeholder="Password" placeholderTextColor="#B3B3B3" secureTextEntry value={password} onChangeText={setPassword} />

      setUser(profile);
    },
    signOut: async () => { await clearSession(); setUser(null); },
    user,
  }), [user]);

  if (!booted) {
    return (
      <View style={{ flex: 1, backgroundColor: '#141414', justifyContent: 'center', alignItems: 'center' }}>
        <ActivityIndicator color="#E50914" />
      </View>
    );
  }

  return (
    <ThemeProvider>
      <NavigationContainer>
        <Animated.View style={[styles.homeWrapper, { opacity: homeOpacity }]}>
          <Stack.Navigator screenOptions={{ headerStyle: { backgroundColor: '#141414' }, headerTintColor: '#fff', contentStyle: { backgroundColor: '#141414' } }}>
            {user ? (
              <>
                <Stack.Screen name="Home" options={{ title: 'StreamNow' }}>{(props) => <Home {...props} auth={auth} />}</Stack.Screen>
                <Stack.Screen name="Player" options={{ title: 'Now Playing' }}>{(props) => <Player {...props} />}</Stack.Screen>
              </>

              <Stack.Screen name="SignIn" options={{ title: 'Sign In' }}>{(props) => <SignIn {...props} auth={auth} />}</Stack.Screen>

          </Stack.Navigator>
        </Animated.View>
        {showSplash && <SplashScreen onFinish={() => setShowSplash(false)} />}
      </NavigationContainer>
    </ThemeProvider>

import { Linking } from 'react-native';

export async function openYouTube(videoIdOrUrl) {
  const id = videoIdOrUrl.includes('youtube.com')
    ? new URL(videoIdOrUrl).searchParams.get('v')
    : videoIdOrUrl;
  const appUrl = `youtube://www.youtube.com/watch?v=${id}`;
  const webUrl = `https://www.youtube.com/watch?v=${id}`;

  const supported = await Linking.canOpenURL(appUrl);
  if (supported) return Linking.openURL(appUrl);
  return Linking.openURL(webUrl);
}

Google Sign-In for bare React Native on iOS and Android
Below are exact platform steps and code to integrate Google Sign-In in a bare React Native app using @react-native-google-signin/google-signin, including native configuration, token handling, and sign-in/sign-out flows.

Overview
Library: @react-native-google-signin/google-signin

What you‚Äôll set up:

Native iOS configuration with URL Types and iOS OAuth client

Native Android configuration with package name and SHA-1

Google Cloud Console OAuth credentials

React Native code to initialize, sign in, and sign out

Secure token storage in Keychain/Keystore

Prerequisites
Google Cloud Console project: Create or use an existing project.

Bundle identifier: e.g., com.yourcompany.streamnow  (iOS).

Package name: e.g., com.yourcompany.streamnow  (Android).

SHA-1 fingerprints: Debug and release for Android.

iOS setup
1. Install library and pods
bash
npm i @react-native-google-signin/google-signin
cd ios && pod install && cd ..
2. Create iOS OAuth client ID
In Google Cloud Console:

Go to ‚ÄúCredentials‚Äù ‚Üí ‚ÄúCreate credentials‚Äù ‚Üí ‚ÄúOAuth client ID‚Äù.

Application type: iOS.

Bundle ID: Your app‚Äôs bundle (e.g., com.yourcompany.streamnow).

Save the client. Copy the generated iOS Client ID and its Reversed client ID (looks like com.googleusercontent.apps.xxxxx).

3. Add URL Type (Info.plist)
Open Xcode ‚Üí select your app target ‚Üí ‚ÄúInfo‚Äù ‚Üí ‚ÄúURL Types‚Äù ‚Üí ‚Äú+‚Äù.

Identifier: GoogleSignIn

URL Schemes: Paste the Reversed client ID (e.g., com.googleusercontent.apps.123456-abcxyz)

Alternatively, edit Info.plist  XML:

xml
<key>CFBundleURLTypes</key>
<array>
  <dict>
    <key>CFBundleURLSchemes</key>
    <array>
      <string>com.googleusercontent.apps.123456-abcxyz</string>
    </array>
  </dict>
</array>
4. Configure iOS client in code
Use the iOS Client ID in GoogleSignin.configure:

js
GoogleSignin.configure({
  iosClientId: 'YOUR_IOS_CLIENT_ID.apps.googleusercontent.com',
});
Android setup
1. Get SHA-1 fingerprints
Debug SHA-1:

bash
keytool -list -v -keystore ~/.android/debug.keystore -alias androiddebugkey -storepass android -keypass android
Release SHA-1: from your signing keystore:

bash
keytool -list -v -keystore /path/to/your-release-key.keystore -alias your_alias
2. Create Android OAuth client ID
In Google Cloud Console:

‚ÄúCredentials‚Äù ‚Üí ‚ÄúCreate credentials‚Äù ‚Üí ‚ÄúOAuth client ID‚Äù.

Application type: Android.

Package name: com.yourcompany.streamnow..

SHA-1 certificate fingerprint: paste the debug SHA-1 for dev, add release SHA-1 when shipping.

3. Add internet permission (usually present)
Confirm AndroidManifest.xml  includes:

xml
<uses-permission android:name="android.permission.INTERNET" />
4. Configure Android client in code
Use the Web Client ID if you need server-side verification or Firebase:

js
GoogleSignin.configure({
  webClientId: 'YOUR_WEB_CLIENT_ID.apps.googleusercontent.com', // optional but recommended
});
Note: For pure Google Sign-In profile/email without server verification, configuring only Android/iOS client IDs is typically sufficient. If you require ID tokens that verify on a backend, set webClientId.

import { NavigationContainer } from '@react-navigation/native';
import { GoogleSignin } from '@react-native-google-signin/google-signin';
import { createNativeStackNavigator } from '@react-navigation/native-stack';
import SignIn from './src/screens/SignIn';
import Home from './src/screens/Home';
import Player from './src/screens/Player';
import { saveSession, loadSession, clearSession } from './src/services/session'; // AsyncStorage for profile
import { saveToken, clearToken } from './src/services/secure'; // Keychain for tokens
import { ThemeProvider } from './src/theme/ThemeContext';

const Stack = createNativeStackNavigator();

export default function App() {
  const [user, setUser] = useState(null);

  useEffect(() => {
    // Configure Google Signin (both platforms)
    GoogleSignin.configure({
      iosClientId: 'YOUR_IOS_CLIENT_ID.apps.googleusercontent.com',
      webClientId: 'YOUR_WEB_CLIENT_ID.apps.googleusercontent.com', // recommended if you need idToken
      offlineAccess: true,
      forceCodeForRefreshToken: false,
    });

    (async () => {
      const existing = await loadSession();
      if (existing) setUser(existing);
    })();
  }, []);

  const auth = useMemo(() => ({
    signInLocal: async (email, password) => {
      if (!email?.includes('@') || password?.length < 6) throw new Error('Invalid credentials');
      const profile = { email, provider: 'local' };
      await saveSession(profile);
      setUser(profile);
    },
    signInGoogleNative: async () => {
      await GoogleSignin.hasPlayServices({ showPlayServicesUpdateDialog: true });
      const account = await GoogleSignin.signIn(); // opens native UI
      const tokens = await GoogleSignin.getTokens(); // { idToken, accessToken }
      const profile = {
        email: account.user.email,
        name: account.user.name,
        photo: account.user.photo,
        provider: 'google',
      };
      await saveSession(profile);
      if (tokens?.idToken) await saveToken('idToken', tokens.idToken);
      if (tokens?.accessToken) await saveToken('accessToken', tokens.accessToken);
      setUser(profile);
    },
    signOut: async () => {
      try { await GoogleSignin.signOut(); } catch {}
      await clearSession();
      await clearToken('idToken');
      await clearToken('accessToken');
      setUser(null);
    },
    user,
  }), [user]);

  return (
    <ThemeProvider>
      <NavigationContainer>
        <Stack.Navigator screenOptions={{ headerStyle: { backgroundColor: '#141414' }, headerTintColor: '#fff', contentStyle: { backgroundColor: '#141414' } }}>
          {user ? (
            <>
              <Stack.Screen name="Home" options={{ title: 'StreamNow' }}>
                {(props) => <Home {...props} auth={auth} />}
              </Stack.Screen>
              <Stack.Screen name="Player" options={{ title: 'Now Playing' }}>
                {(props) => <Player {...props} />}
              </Stack.Screen>
            </>
          ) : (
            <Stack.Screen name="SignIn" options={{ title: 'Sign In' }}>
              {(props) => <SignIn {...props} auth={auth} />}
            </Stack.Screen>
          )}
        </Stack.Navigator>
      </NavigationContainer>
    </ThemeProvider>
  );
}

// src/screens/SignIn.js
import React from 'react';
import { View, Text, TextInput, TouchableOpacity, StyleSheet, Alert } from 'react-native';

export default function SignIn({ auth }) {
  const [email, setEmail] = React.useState('');
  const [password, setPassword] = React.useState('');

  const onLocal = async () => {
    try { await auth.signInLocal(email, password); }
    catch (e) { Alert.alert('Sign-in failed', e.message); }
  };

  const onGoogleNative = async () => {
    try { await auth.signInGoogleNative(); }
    catch (e) { Alert.alert('Google sign-in failed', e.message); }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Welcome back</Text>

      <TextInput
        style={styles.input}
        placeholder="Email"
        placeholderTextColor="#B3B3B3"
        autoCapitalize="none"
        keyboardType="email-address"
        value={email}
        onChangeText={setEmail}
      />

      <TextInput
        style={styles.input}
        placeholder="Password"
        placeholderTextColor="#B3B3B3"
        secureTextEntry
        value={password}
        onChangeText={setPassword}
      />

      <TouchableOpacity style={styles.cta} onPress={onLocal}>
        <Text style={styles.ctaText}>Sign In</Text>
      </TouchableOpacity>

      <TouchableOpacity style={styles.google} onPress={onGoogleNative}>
        <Text style={styles.googleText}>Continue with Google</Text>
      </TouchableOpacity>
    </View>
  );
}

2. Secure storage helpers
js
// src/services/secure.js
import * as Keychain from 'react-native-keychain';

export async function saveToken(key, value) {
  // key used as service name; value stored as password
  await Keychain.setGenericPassword(key, value, { service: `streamnow.${key}` });
}

export async function getToken(key) {
  const creds = await Keychain.getGenericPassword({ service: `streamnow.${key}` });
  return creds ? creds.password : null;
}

export async function clearToken(key) {
  await Keychain.resetGenericPassword({ service: `streamnow.${key}` });
}
Common platform notes and checks
iOS URL Types: Without the reversed client ID in URL Types, the Google Sign-In callback will fail.

Android SHA-1: Missing or incorrect SHA-1 prevents sign-in on device; add both debug and release fingerprints to your Android OAuth client.

webClientId: Needed if you plan to verify ID tokens on a backend or use Firebase auth flows. Otherwise, basic profile/email can work with platform-specific clients.

Play Services: On Android, ensure Google Play Services are present and updated on test devices.

Scopes: Default scopes provide profile/email. If you need more, add scopes: ['profile', 'email', 'openid'] to configure.

Optional: Sign-in status and silent sign-in
Check if user is signed in:

js
const isSignedIn = await GoogleSignin.isSignedIn();
Try silent sign-in:

js
const userInfo = await GoogleSignin.signInSilently();
Optional: Deep linking and intents
If you later add deep linking for Google or additions to iOS LSApplicationQueriesSchemes, ensure you register appropriate schemes. Basic Google Sign-In via the library typically does not require extra LSApplicationQueriesSchemes.

Quick checklist
Created iOS OAuth client with correct bundle ID.

Added URL Type with reversed client ID in Info.plist..

Created Android OAuth client with package name and SHA-1(s).

Configured GoogleSignin with iosClientId and optional webClientId.

Integrated sign-in/sign-out with secure token storage.

Tested on a physical device/emulator for both platforms.
