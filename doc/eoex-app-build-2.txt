import React from 'react';
import { View, Text, FlatList, Image, StyleSheet } from 'react-native';

const posters = [
  { id: '1', title: 'Sample 1', uri: 'https://via.placeholder.com/200x300' },
  { id: '2', title: 'Sample 2', uri: 'https://via.placeholder.com/200x300' },
  { id: '3', title: 'Sample 3', uri: 'https://via.placeholder.com/200x300' },
];

export default function Carousel({ title }) {
  return (
    <View style={styles.container}>
      <Text style={styles.title}>{title}</Text>
      <FlatList
        horizontal
        showsHorizontalScrollIndicator={false}
        data={posters}
        keyExtractor={(item) => item.id}
        renderItem={({ item }) => (
          <View style={styles.card}>
            <Image source={{ uri: item.uri }} style={styles.poster} />
            <Text style={styles.caption}>{item.title}</Text>
          </View>
        )}
      />
    </View>
  );
}

const styles = StyleSheet.create({
  container: { padding: 16 },
  title: { color: '#fff', fontSize: 20, fontWeight: '700', marginBottom: 12 },
  card: { marginRight: 12 },
  poster: { width: 120, height: 180, borderRadius: 8 },
  caption: { color: '#B3B3B3', marginTop: 4, fontSize: 12 },
});
FAQ.js (Accordion)
jsx
import React, { useState } from 'react';
import { View, Text, TouchableOpacity, LayoutAnimation, StyleSheet } from 'react-native';

const faqs = [
  { q: 'What is StreamNow?', a: 'Stream a wide variety of shows and movies on your devices.' },
  { q: 'How much does it cost?', a: 'Plans start at â‚¬7.99 per month.' },
];

export default function FAQ() {
  const [openIndex, setOpenIndex] = useState(null);

  const toggle = (i) => {
    LayoutAnimation.configureNext(LayoutAnimation.Presets.easeInEaseOut);
    setOpenIndex(openIndex === i ? null : i);
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Frequently Asked Questions</Text>
      {faqs.map((item, i) => (
        <View key={i} style={styles.item}>
          <TouchableOpacity onPress={() => toggle(i)}>
            <Text style={styles.question}>{item.q}</Text>
          </TouchableOpacity>
          {openIndex === i && <Text style={styles.answer}>{item.a}</Text>}
        </View>
      ))}
    </View>
  );
}

const styles = StyleSheet.create({
  container: { padding: 16 },
  title: { color: '#fff', fontSize: 20, fontWeight: '700', marginBottom: 12 },
  item: { marginBottom: 12 },
  question: { color: '#fff', fontSize: 16, fontWeight: '600' },
  answer: { color: '#B3B3B3', marginTop: 6 },
});
