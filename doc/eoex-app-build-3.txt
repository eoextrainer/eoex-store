React Native Netflix-style app with navigation, auth, and optimized media
I’m keeping the soul of the dark, bold aesthetic while wiring in navigation, simple auth, Google sign-in, FastImage posters, and lazy-loaded sections. The YouTube items open in a dedicated Player screen.

Tech stack and packages
Core: React Native (Expo or bare RN), React Navigation

Media & performance: react-native-fast-image (posters), FlatList (virtualized, lazy rendering)

UI: react-native-linear-gradient (hero), react-native-vector-icons (optional chevrons/icons)

Auth:

Username/password: Local validation for demo

Google: Use expo-auth-session (Expo) or @react-native-google-signin/google-signin (bare RN)

If you’re using Expo, prefer expo-auth-session. For bare RN, use @react-native-google-signin/google-signin.

App structure
Screens: SignIn, Home, Player

Components: Hero, MediaCarousel, Features, FAQ, Footer

Data: Static YouTube items with derived thumbnails

Navigation and auth scaffold
jsx
// App.js
import React, { useMemo, useState, useEffect } from 'react';
import { NavigationContainer } from '@react-navigation/native';
import { createNativeStackNavigator } from '@react-navigation/native-stack';
import SignIn from './src/screens/SignIn';
import Home from './src/screens/Home';
import Player from './src/screens/Player';

const Stack = createNativeStackNavigator();

export default function App() {
  const [user, setUser] = useState(null);

  const auth = useMemo(() => ({
    signInLocal: async (email, password) => {
      // Demo-only: simple validation
      if (!email?.includes('@') || password?.length < 6) throw new Error('Invalid credentials');
      setUser({ email });
    },
    signOut: () => setUser(null),
    signInGoogle: async (profile) => {
      setUser(profile);
    },
    user,
  }), [user]);

  return (
    <NavigationContainer>
      <Stack.Navigator
        screenOptions={{
          headerStyle: { backgroundColor: '#141414' },
          headerTintColor: '#fff',
          contentStyle: { backgroundColor: '#141414' },
        }}
      >
        {user ? (
          <>
            <Stack.Screen name="Home" options={{ title: 'StreamNow' }}>
              {(props) => <Home {...props} auth={auth} />}
            </Stack.Screen>
            <Stack.Screen name="Player" options={{ title: 'Now Playing' }}>
              {(props) => <Player {...props} />}
            </Stack.Screen>
          </>
        ) : (
          <Stack.Screen name="SignIn" options={{ title: 'Sign In' }}>
            {(props) => <SignIn {...props} auth={auth} />}
          </Stack.Screen>
        )}
      </Stack.Navigator>
    </NavigationContainer>
  );
}
Sign-in screen with local and Google
Expo using expo-auth-session
jsx
// src/screens/SignIn.js
import React, { useEffect } from 'react';
import { View, Text, TextInput, TouchableOpacity, StyleSheet, Alert } from 'react-native';
import * as WebBrowser from 'expo-web-browser';
import * as AuthSession from 'expo-auth-session';

WebBrowser.maybeCompleteAuthSession();

const googleConfig = {
  clientId: '<YOUR_GOOGLE_OAUTH_CLIENT_ID>',
  responseType: 'token',
  scopes: ['profile', 'email'],
  redirectUri: AuthSession.makeRedirectUri({ useProxy: true }),
};

export default function SignIn({ auth }) {
  const [email, setEmail] = React.useState('');
  const [password, setPassword] = React.useState('');

  const onLocal = async () => {
    try {
      await auth.signInLocal(email, password);
    } catch (e) {
      Alert.alert('Sign-in failed', e.message);
    }
  };

  const onGoogle = async () => {
    const authUrl = `https://accounts.google.com/o/oauth2/v2/auth?client_id=${googleConfig.clientId}&redirect_uri=${encodeURIComponent(googleConfig.redirectUri)}&response_type=${googleConfig.responseType}&scope=${encodeURIComponent(googleConfig.scopes.join(' '))}`;
    const result = await AuthSession.startAsync({ authUrl });
    if (result.type === 'success' && result.params?.access_token) {
      // For demo: you would fetch profile from Google APIs using the token
      auth.signInGoogle({ email: 'google_user@example.com' });
    } else {
      Alert.alert('Google sign-in cancelled');
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Welcome back</Text>
      <TextInput
        style={styles.input}
        placeholder="Email"
        placeholderTextColor="#B3B3B3"
        autoCapitalize="none"
        keyboardType="email-address"
        value={email}
        onChangeText={setEmail}
      />
      <TextInput
        style={styles.input}
        placeholder="Password"
        placeholderTextColor="#B3B3B3"
        secureTextEntry
        value={password}
        onChangeText={setPassword}
      />
      <TouchableOpacity style={styles.cta} onPress={onLocal}>
        <Text style={styles.ctaText}>Sign In</Text>
      </TouchableOpacity>

      <TouchableOpacity style={styles.google} onPress={onGoogle}>
        <Text style={styles.googleText}>Continue with Google</Text>
      </TouchableOpacity>
    </View>
  );
}

const styles = StyleSheet.create({
  container: { padding: 24 },
  title: { color: '#fff', fontSize: 24, fontWeight: '700', marginBottom: 16 },
  input: { backgroundColor: '#221F1F', color: '#fff', padding: 12, borderRadius: 8, marginBottom: 12 },
  cta: { backgroundColor: '#E50914', padding: 12, borderRadius: 8, alignItems: 'center', marginTop: 8 },
  ctaText: { color: '#fff', fontWeight: '600' },
  google: { backgroundColor: '#fff', padding: 12, borderRadius: 8, alignItems: 'center', marginTop: 12 },
  googleText: { color: '#000', fontWeight: '600' },
});

Home screen with hero, lazy carousels, features, FAQ
jsx
// src/screens/Home.js
import React, { useRef } from 'react';
import { View, Text, StyleSheet, FlatList } from 'react-native';
import Hero from '../components/Hero';
import MediaCarousel from '../components/MediaCarousel';
import Features from '../components/Features';
import FAQ from '../components/FAQ';
import Footer from '../components/Footer';
import { videos } from '../data/videos';

const sections = [
  { key: 'hero' },
  { key: 'trending', title: 'Trending Now', data: videos.slice(0, 5) },
  { key: 'more', title: 'Must Watch', data: videos.slice(5) },
  { key: 'features' },
  { key: 'faq' },
  { key: 'footer' },
];

export default function Home({ navigation, auth }) {
  // Lazy-render sections via FlatList virtualization
  return (
    <FlatList
      data={sections}
      keyExtractor={(item) => item.key}
      renderItem={({ item }) => {
        switch (item.key) {
          case 'hero':
            return <Hero />;
          case 'trending':
          case 'more':
            return (
              <MediaCarousel
                title={item.title}
                items={item.data}
                onPressItem={(video) => navigation.navigate('Player', { video })}
              />
            );
          case 'features':
            return <Features />;
          case 'faq':
            return <FAQ />;
          case 'footer':
            return <Footer onSignOut={auth.signOut} />;
          default:
            return null;
        }
      }}
      contentContainerStyle={{ paddingBottom: 24 }}
      initialNumToRender={2}
      windowSize={5}
      removeClippedSubviews
    />
  );
}

// src/components/MediaCarousel.js
import React, { useMemo } from 'react';
import { View, Text, StyleSheet, FlatList, TouchableOpacity } from 'react-native';
import FastImage from 'react-native-fast-image';
import { getThumbnail } from '../data/videos';

export default function MediaCarousel({ title, items, onPressItem }) {
  const data = useMemo(() => items, [items]);

  return (
    <View style={styles.container}>
      <Text style={styles.title}>{title}</Text>
      <FlatList
        horizontal
        data={data}
        keyExtractor={(item) => item.id}
        showsHorizontalScrollIndicator={false}
        initialNumToRender={4}
        windowSize={7}
        renderItem={({ item }) => (
          <TouchableOpacity onPress={() => onPressItem(item)} style={styles.card} activeOpacity={0.8}>
            <FastImage
              style={styles.poster}
              source={{
                uri: getThumbnail(item.id),
                priority: FastImage.priority.normal,
              }}
              resizeMode={FastImage.resizeMode.cover}
            />
            <Text style={styles.caption} numberOfLines={1}>{item.title}</Text>
          </TouchableOpacity>
        )}
      />
    </View>
  );
}

const styles = StyleSheet.create({
  container: { paddingVertical: 16, paddingHorizontal: 16 },
  title: { color: '#fff', fontSize: 20, fontWeight: '700', marginBottom: 12 },
  card: { marginRight: 12, width: 140 },
  poster: { width: 140, height: 210, borderRadius: 8, backgroundColor: '#221F1F' },
  caption: { color: '#B3B3B3', marginTop: 6, fontSize: 12 },

// src/components/FAQ.js
import React, { useState } from 'react';
import { View, Text, TouchableOpacity, LayoutAnimation, StyleSheet } from 'react-native';

export default function FAQ() {
  const [open, setOpen] = useState(null);
  const items = [
    { q: 'What is StreamNow?', a: 'Stream shows and movies on your devices.' },
    { q: 'How much does it cost?', a: 'Plans start at €7.99 per month.' },
  ];
  const toggle = (i) => {
    LayoutAnimation.configureNext(LayoutAnimation.Presets.easeInEaseOut);
    setOpen(open === i ? null : i);
  };
  return (
    <View style={styles.container}>
      <Text style={styles.title}>Frequently Asked Questions</Text>
      {items.map((it, i) => (
        <View key={i} style={styles.item}>
          <TouchableOpacity onPress={() => toggle(i)}>
            <Text style={styles.q}>{it.q}</Text>
          </TouchableOpacity>
          {open === i && <Text style={styles.a}>{it.a}</Text>}
        </View>
      ))}
    </View>
  );
}
const styles = StyleSheet.create({
  container: { padding: 16 },
  title: { color: '#fff', fontSize: 20, fontWeight: '700', marginBottom: 12 },

Performance notes and tweaks
FastImage caching: Use FastImage for posters in production; set priority and consider cache strategy.

Lazy sections: Home uses FlatList to virtualize sections; tune initialNumToRender, windowSize, and removeClippedSubviews.

Batch updates: Memoize props for carousels; avoid re-renders by stabilizing data arrays.

Assets: Derive YouTube poster thumbnails via img.youtube.com to avoid additional APIs.

Brand kit (for consistency)
Colors: Primary red #E50914; background black #000000; dark gray #141414; panel #221F1F; white #FFFFFF; secondary gray #B3B3B3.

Typography: Bold headlines 28–32px; medium subheads 18–20px; body 16px; system sans or Inter.

Components: Red CTAs with white text and 8–12px radius; dark panels; visible focus/press states.
